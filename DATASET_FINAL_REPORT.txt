Load Dataset in Python:

```python
import cv2
import pandas as pd
from pathlib import Path

# Load annotations
annotations = pd.read_csv("dataset_ir_realistic/annotations.csv")

# Load an image
img = cv2.imread("dataset_ir_realistic/images/img_000000.png",
                 cv2.IMREAD_GRAYSCALE)

# Access features
row = annotations[annotations['filename'] == 'img_000000.png'].iloc[0]
print(f"Label: {row['label']}")  # 1 (bullet), 0 (non), -1 (ambiguous)
print(f"Center: ({row['center_x']}, {row['center_y']})")
print(f"Radius: {row['radius']}")
```
